141  LCD_TXT_CENTER, LCD_UNITS_COLOR);
142  } else {
143

145  // select decimal places, then write the units to the units line on the LCD.

267 /**
268  * \brief Get the resistance value.
269  * \remarks Reads the A/D converter to get raw data then converts it to a resistance. The
270  * resistance value is filtered with a digital filter and the difference between the
271  * computed resistance and the filtered resistance is squared and filtered. The filters
272  * implement simple linear recursive exponential filters. These are essentially first
273  * order FIR filters which are ideal for the removal of white noise typically found on
274  * sensor data. They are not suitable for spectral filtering.
275  * \par
276  * The meter circuit provides 100mA of source current so the resistance
277  * is equal to Measured Voltage * 10 where Measured Voltage is the raw a/d value
278  * multiplied by a gain factor in V/bit.
279  * \par
280  * The raw a/d value is compared to the hysteresis limit for the purpose of changing
281  * the gain of the PGA. If the raw value is below the lower limit then the gain is
282  * increased. If the value is above the upper limit then the gain is reduced. The
283  * hysteresis is necessary to prevent the display from flickering between ranges.
284  */
285 void getResistance() {
286 
287  // Get the raw integer value of the a/d converter. Valide range is 0..0x7FFF
288  int16_t raw = Ads1115.readADC_SingleEnded(0);
289 
290  #ifdef _DEBUG_
291  rawFilter.Filter(raw);
292  rw = raw;
293  gn = Ads1115.getGain();
294  #endif // _DEBUG_
295 
296  // Just throw away negative voltages.
297  if(raw < 0) raw = 0;
298 
299  // Use this value to update the resistance and deviation filter.
300  // adjustGain() returns Ohms/bit.
301  double res = adjustGain(DIR_AS) * raw;
302  resSFilter.Filter(res);
303  resHFilter.Filter(res);
304  devFilter.Filter(sqr(res-resHFilter.Current()));
305 
306  // Change the gain settings for next loop based on specified hysteresis limits.
307  if (raw < RAW_ADS_SPAN/2-RAW_HYSTERESIS) adjustGain(DIR_UP);
308  else if(raw > RAW_ADS_SPAN/2+RAW_HYSTERESIS) adjustGain(DIR_DN);
309 }

311 /**
312  * \brief Change the gain of the A/D converter PGA.
313  * \remarks Changes the gain by one step only, either up, down, or no change.
314  * \param [in] Direction_t dir Direction to adjust the gain.
315  * \return double gain factor (Ohms/bit).
316  */
317 double adjustGain(Direction_t dir) {
318 
319  adsGain_t gain = Ads1115.getGain();
320 
321  if(dir != DIR_AS) {
322  switch (gain) {
323  case GAIN_SIXTEEN : gain = dir == DIR_UP ? GAIN_SIXTEEN : GAIN_EIGHT; break;
324  case GAIN_EIGHT : gain = dir == DIR_UP ? GAIN_SIXTEEN : GAIN_FOUR; break;
325  case GAIN_FOUR : gain = dir == DIR_UP ? GAIN_EIGHT : GAIN_TWO; break;
326  case GAIN_TWO : gain = dir == DIR_UP ? GAIN_FOUR : GAIN_ONE; break;
327  case GAIN_ONE : gain = dir == DIR_UP ? GAIN_TWO : GAIN_TWOTHIRDS; break;
328  default : gain = dir == DIR_UP ? GAIN_ONE : GAIN_TWOTHIRDS;
329  }
330  Ads1115.setGain(gain);
331  }
332  // Return the gain in Ohms/bit.
333  switch (gain) {
334  case GAIN_SIXTEEN : return VPB_SIXTEEN / RES_TEST_CURRENT / DIFF_AMP_GAIN;
335  case GAIN_EIGHT : return VPB_EIGHT / RES_TEST_CURRENT / DIFF_AMP_GAIN;
336  case GAIN_FOUR : return VPB_FOUR / RES_TEST_CURRENT / DIFF_AMP_GAIN;
337  case GAIN_TWO : return VPB_TWO / RES_TEST_CURRENT / DIFF_AMP_GAIN;
338  case GAIN_ONE : return VPB_ONE / RES_TEST_CURRENT / DIFF_AMP_GAIN;
339  }
340  return VPB_TWOTHIRDS / RES_TEST_CURRENT / DIFF_AMP_GAIN; // default is GAIN_TWOTHIRDS
341 }

356 /**
357  * \brief Get the battery voltage.
358  * \remarks The battery is a single LiIon cylindrical cell. It is nominally 3.7V and 4.2V at
359  * full charge. It is 2.4V fully discharged. The voltage is measured single ended
360  * (ie. referenced to ground) and the gain is set for a 6.144V range. On completion
361  * the gain is restored to its previous value.
362  */
363 void getBattVolts() {
364 
365  // Save the A/D converter gain.
366  adsGain_t gain = Ads1115.getGain();
367  Ads1115.setGain(GAIN_TWOTHIRDS);
368 
369  // Get the battery voltage using the lowest A/D converter gain..
370  dblBatV = VPB_TWOTHIRDS * Ads1115.readADC_SingleEnded(2);
371 
372  // Restore the A/D converter gain.
373  Ads1115.setGain(gain);
374 
375  // Convert the battery voltage to a state of charge (0 to 200, 0.5% increments).
376  uint8_t chgVal = dblBatV < BAT_VDISCHARGED ? 0 : dblBatV >= BAT_VCHARGED ? 200 :
377  (uint8_t)((dblBatV-BAT_VDISCHARGED)/(BAT_VCHARGED-BAT_VDISCHARGED)*200.0);
378 
379  // Get the battery operating status from the status pins.
380  bool chg = !digitalRead(CHG_PIN);
381  bool stby = !digitalRead(STBY_PIN);
382 
383  // Convert the operating status to a battery state.
384  BatState chgState;
385  if(chg && !stby) chgState = BAT_CHARGING;
386  else if((chgVal < 200) && (chgVal > 0)) chgState = BAT_NORMAL;
387  else if (!chg && stby && (chgVal >= 200)) chgState = BAT_FULL;
388  else if (!chg && !stby) chgState = BAT_NONE;
389  else chgState = BAT_EMPTY;
390 
391  // Now draw the battery indicator showing battery state and voltage.
392  drawBatteryStatus(chgVal, chgState);
393 }
